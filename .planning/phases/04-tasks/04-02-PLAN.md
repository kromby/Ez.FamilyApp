```typescript
// Task types
export interface Task {
  id: string;
  familyId: string;
  name: string;
  addedById: string;
  addedByName: string;
  completedAt: string | null;
  completedById: string | null;
  completedByName: string | null;
  createdAt: string;
}

export async function fetchTasks(token: string): Promise<{ tasks: Task[] }> {
  return apiFetch('/tasks', token);
}

export async function addTask(token: string, name: string): Promise<{ task: Task }> {
  return apiFetch('/tasks', token, {
    method: 'POST',
    body: JSON.stringify({ name }),
  });
}

export async function toggleTask(token: string, taskId: string, completed: boolean): Promise<{ task: Task }> {
  return apiFetch(`/tasks/${taskId}`, token, {
    method: 'PATCH',
    body: JSON.stringify({ completed }),
  });
}

export async function deleteTask(token: string, taskId: string): Promise<void> {
  await apiFetch(`/tasks/${taskId}`, token, { method: 'DELETE' });
}
```

**3. Create `src/hooks/useTasks.ts`:**

TanStack Query hooks with optimistic updates:

- `useTasks()`: `useQuery({ queryKey: ['tasks'], queryFn: () => fetchTasks(token), enabled: !!token })`. Returns `{ tasks, isLoading, refetch }`. The tasks array should be returned as-is from the API (already sorted by created_at DESC on server).

- `useAddTask()`: `useMutation` calling `addTask(token, name)`. `onMutate`: create a temp task with `id: 'temp-' + Date.now()`, `addedById: userId`, `addedByName: displayName` (get from session), `completedAt: null`, `createdAt: new Date().toISOString()`. Prepend to `['tasks']` cache. `onError`: rollback via context. `onSuccess`: replace temp with real task in cache by matching `(addedById + name)` for the temp record, or invalidate queries.

- `useToggleTask()`: `useMutation` calling `toggleTask(token, taskId, completed)`. `onMutate`: optimistically update the task in `['tasks']` cache — set `completedAt` to `new Date().toISOString()` and `completedByName` to current user's displayName if completing, or set all three to null if unchecking. `onError`: rollback. `onSuccess`: update cache with server response (full task record).

- `useDeleteTask()`: `useMutation` calling `deleteTask(token, taskId)`. `onMutate`: optimistically remove from `['tasks']` cache. `onError`: rollback (restore task). `onSuccess`: no-op (already removed). Handle 404 errors silently per D-15 (catch error, check if message contains "404", treat as success).

Import `useSession` to get `token`, `userId`. Get `displayName` from session store for optimistic payloads.

**1. handleTaskAdded callback:**
```typescript
const handleTaskAdded = useCallback((task: Task) => {
  queryClient.setQueryData(['tasks'], (old: { tasks: Task[] } | undefined) => {
    if (!old) return { tasks: [task] };
    // Dedup: skip if task.id already exists OR if a temp task matches (addedById + name)
    const exists = old.tasks.some(
      (t) => t.id === task.id || (t.id.startsWith('temp-') && t.addedById === task.addedById && t.name === task.name)
    );
    if (exists) {
      // Replace temp with real
      return {
        tasks: old.tasks.map((t) =>
          t.id.startsWith('temp-') && t.addedById === task.addedById && t.name === task.name ? task : t
        ),
      };
    }
    return { tasks: [task, ...old.tasks] };
  });
}, [queryClient]);
```

**2. handleTaskUpdated callback:**
```typescript
const handleTaskUpdated = useCallback((task: Task) => {
  queryClient.setQueryData(['tasks'], (old: { tasks: Task[] } | undefined) => {
    if (!old) return old;
    return { tasks: old.tasks.map((t) => (t.id === task.id ? task : t)) };
  });
}, [queryClient]);
```

**3. handleTaskDeleted callback:**
```typescript
const handleTaskDeleted = useCallback((payload: { taskId: string }) => {
  queryClient.setQueryData(['tasks'], (old: { tasks: Task[] } | undefined) => {
    if (!old) return old;
    return { tasks: old.tasks.filter((t) => t.id !== payload.taskId) };
  });
}, [queryClient]);
```
